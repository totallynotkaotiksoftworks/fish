local function loadScript(url)
    local success, result = pcall(loadstring(game:HttpGet(url)))
    if not success then
        warn("Failed to load script from " .. url .. ": " .. result)
    end
end

local scripts = {
    "https://raw.githubusercontent.com/totallynotkaotiksoftworks/fish/refs/heads/main/utilities.lua",
    "https://raw.githubusercontent.com/totallynotkaotiksoftworks/fish/refs/heads/main/settings.lua",
    "https://raw.githubusercontent.com/totallynotkaotiksoftworks/fish/refs/heads/main/ui.lua",
    "https://raw.githubusercontent.com/totallynotkaotiksoftworks/fish/refs/heads/main/core.lua"
}

for _, url in ipairs(scripts) do
    loadScript(url)
end
